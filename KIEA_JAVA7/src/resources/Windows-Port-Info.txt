[ 원격 테스트톱 포트 변경 배치 프로그램 ]

	@echo off
	title 원격 데스크톱 포트 변경
	cls
	echo.
	set /p port=변경하고 싶은 새 포트를 입력해주세요: 
	cls
	echo.
	echo 요청하신 포트는 %port% 에요. 계속 하시려면 아무 키나 누르시고, 아니라면 창을 닫아주세요.
	echo.
	pause
	reg add "HKLM\SYSTEM\CurrentControlSet\Control\Terminal Server" /v fDenyTSConnections /t REG_DWORD /d 0 /f
	reg add "HKLM\SYSTEM\CurrentControlSet\Control\Terminal Server\WinStations\RDP-Tcp" /v PortNumber /t REG_DWORD /d %port% /f
	if errorlevel 1 (
		cls
		echo.
		echo 관리자 권한으로 실행해주세요.
		echo.
		pause>nul
		exit
	)
	netsh advfirewall firewall add rule name="Remote Desktop" dir=in action=allow protocol=tcp localport=%port%
	net stop /y TermService
	net start /y TermService
	cls
	echo.
	echo 작업이 끝났어요.
	echo.
	pause
	exit
